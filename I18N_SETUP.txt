USAGE EXAMPLES
==============

Server Component (Next.js 14+):
--
import { t, buildMetadata } from '@/lib/i18n';
import es from '@/translations/es.json';

export const metadata = buildMetadata({
  lang: 'es',
  title: 'CDMX Metro Guide',
  description: 'Transit guide for Mexico City',
  path: '/cities/cdmx',
  keywords: ['metro', 'transit']
});

export default function Page() {
  const heroTitle = t(es, 'home.heroTitle');
  return <h1>{heroTitle}</h1>;
}

Client Component (Dynamic Switching):
--
'use client';
import { t, LANGUAGE_NAMES } from '@/lib/i18n';
import { useState } from 'react';

const allTrans = {
  es: require('@/translations/es.json'),
  en: require('@/translations/en.json'),
  // ... others
};

export default function LangSwitcher() {
  const [lang, setLang] = useState('es');
  return (
    <>
      <select onChange={(e) => setLang(e.target.value)}>
        {Object.entries(LANGUAGE_NAMES).map(([code, name]) => (
          <option key={code} value={code}>{name}</option>
        ))}
      </select>
      <h1>{t(allTrans[lang], 'home.heroTitle')}</h1>
    </>
  );
}
